// Definisi class sederhana untuk Mahasiswa
class Mahasiswa {
  // Properties (attributes)
  String nama;
  String nim;
  int semester;
  double ipk;
  
  // Constructor
  Mahasiswa(this.nama, this.nim, this.semester, this.ipk);
  
  // Named constructor untuk mahasiswa baru
  Mahasiswa.mahasiswaBaru(String nama, String nim) 
    : this.nama = nama,
      this.nim = nim,
      this.semester = 1,
      this.ipk = 0.0;
  
  // Methods (functions)
  void tampilkanInfo() {
    print('=== INFO MAHASISWA ===');
    print('Nama: $nama');
    print('NIM: $nim');
    print('Semester: $semester');
    print('IPK: ${ipk.toStringAsFixed(2)}');
  }
  
  // Method untuk update IPK
  void updateIPK(double ipkBaru) {
    if (ipkBaru >= 0.0 && ipkBaru <= 4.0) {
      ipk = ipkBaru;
      print('IPK berhasil diupdate menjadi: ${ipk.toStringAsFixed(2)}');
    } else {
      print('Error: IPK harus antara 0.0 - 4.0');
    }
  }
  
  // Method untuk naik semester
  void naikSemester() {
    if (semester < 8) {
      semester++;
      print('Selamat! $nama naik ke semester $semester');
    } else {
      print('$nama sudah di semester maksimal');
    }
  }
  
  // Getter untuk status kelulusan
  String get statusKelulusan {
    if (semester >= 8 && ipk >= 2.0) {
      return 'Lulus';
    } else if (semester >= 8 && ipk < 2.0) {
      return 'Tidak Lulus';
    } else {
      return 'Masih Kuliah';
    }
  }
  
  // Method untuk menentukan predikat
  String getPredikat() {
    if (ipk >= 3.75) return 'Summa Cum Laude';
    if (ipk >= 3.50) return 'Magna Cum Laude';
    if (ipk >= 3.25) return 'Cum Laude';
    if (ipk >= 2.75) return 'Sangat Memuaskan';
    if (ipk >= 2.00) return 'Memuaskan';
    return 'Tidak Memenuhi Syarat';
  }
}

}

Inheritance dan Polymorphism

Inheritance (Pewarisan)

Inheritance memungkinkan sebuah class (child/subclass) untuk mewarisi properties dan methods dari class lain (parent/superclass). Ini membantu dalam code reusability dan menciptakan hierarki yang logical.

Inheritance.dart 
// Parent class (Superclass)
abstract class Person {
  String nama;
  int umur;
  String email;
  
  Person(this.nama, this.umur, this.email);
  
  // Abstract method - harus diimplementasi di child class
  void tampilkanPeran();
  
  // Concrete method - bisa digunakan langsung
  void tampilkanInfoDasar() {
    print('Nama: $nama');
    print('Umur: $umur tahun');
    print('Email: $email');
  }
  
  void kirimEmail(String pesan) {
    print('Email terkirim ke $email: $pesan');
  }
}

void main() {
  // === DEKLARASI LIST ===
  
  // List dengan tipe data eksplisit
  List mahasiswaIF = ['Ahmad', 'Budi', 'Citra', 'Dewi'];
  List nilaiUjian = [85, 92, 78, 95, 88];
  List ipkMahasiswa = [3.4, 3.8, 3.2, 3.9, 3.6];

  List> databaseMahasiswa = [];

  print('Daftar mahasiswa: $mahasiswaIF');
  
  // Mengakses element

  print('Mahasiswa terakhir: ${mahasiswaIF.last}');
  print('Jumlah mahasiswa: ${mahasiswaIF.length}');

  mahasiswaIF.insert(2, 'Hani'); // Insert di index 2

  print('Rata-rata IPK: ${rataRataIPK.toStringAsFixed(2)}');
  
  // === SORTING ===

  nilaiCopy.sort(); // Ascending
  print('\nNilai terurut ascending: $nilaiCopy');
  
  nilaiCopy.sort((a, b) => b.compareTo(a)); // Descending
  print('Nilai terurut descending: $nilaiCopy');
  
  // === LIST COMPREHENSION STYLE ===
  
  // Membuat list mahasiswa berprestasi

  for (int i = 0; i < mahasiswaIF.length && i < ipkMahasiswa.length; i++) {

  }
}

  Map>> mahasiswaPerJurusan = {};
  
  dataMahasiswa.forEach((nim, data) {
    String jurusan = data['jurusan'];
    if (!mahasiswaPerJurusan.containsKey(jurusan)) {

    mahasiswaPerJurusan[jurusan]!.add(data);

    print('\n$jurusan (${listMahasiswa.length} mahasiswa):');
    for (var mahasiswa in listMahasiswa) {
      print('  - ${mahasiswa['nama']} (IPK: ${mahasiswa['ipk']})');
    }

  // === NESTED MAP OPERATIONS ===

    print('\n$mataKuliah:');

  });

void main() {
  // === DEKLARASI SET ===
  
  // Set dengan data initial
  Set mataKuliahWajib = {
    'Algoritma Pemrograman',
    'Struktur Data',
    'Database',
    'Pemrograman Web',
    'Mobile Programming'
  };
  
  Set mataKuliahPilihan = {
    'Artificial Intelligence',
    'Machine Learning',
    'Cyber Security',
    'Game Development',
    'Mobile Programming' // Duplikasi akan diabaikan
  };
  
  // Set kosong
  Set mataKuliahDiambil = {};
  Set mahasiswaAktif = {};
  
  print('=== OPERASI DASAR SET ===');
  
  // Menambah element
  mataKuliahDiambil.add('Database');
  mataKuliahDiambil.add('Mobile Programming');
  mataKuliahDiambil.add('Database'); // Duplikasi diabaikan
  
  print('Mata kuliah diambil: $mataKuliahDiambil');
  print('Jumlah mata kuliah: ${mataKuliahDiambil.length}');
  
  // Mengecek keberadaan element
  bool adaDatabase = mataKuliahDiambil.contains('Database');
  print('Sudah ambil Database? $adaDatabase');
  
  // === SET OPERATIONS ===
  
  print('\n=== OPERASI SET ===');
  
  // Union (gabungan)
  Set semuaMataKuliah = mataKuliahWajib.union(mataKuliahPilihan);
  print('Semua mata kuliah tersedia: ${semuaMataKuliah.length}');
  semuaMataKuliah.forEach((mk) => print('- $mk'));
  
  // Intersection (irisan)

  // Difference (selisih)

  // Simulasi mahasiswa dan mata kuliah yang diambil

class MahasiswaManager {
  Map> _dataMahasiswa = {};
  
  // Validasi format NIM
  bool _isValidNIM(String nim) {
    // NIM harus 10 digit dan dimulai dengan 20 (tahun 2020+)
    RegExp nimPattern = RegExp(r'^20\d{8}$');
    return nimPattern.hasMatch(nim);
  }
  
  // Validasi IPK
  bool _isValidIPK(double ipk) {
    return ipk >= 0.0 && ipk <= 4.0;
  }
  
  // Menambah mahasiswa dengan validation
  void tambahMahasiswa({
    required String nim,
    required String nama,
    required String jurusan,
    required int semester,
    required double ipk,
  }) {
    try {
      // Validasi NIM
      if (!_isValidNIM(nim)) {
        throw InvalidNIMException('Format NIM tidak valid. NIM harus 10 digit dan dimulai dengan 20');
      }
      
      // Validasi IPK
      if (!_isValidIPK(ipk)) {
        throw InvalidIPKException('IPK harus antara 0.0 - 4.0, diberikan: $ipk');
      }
      
      // Validasi semester
      if (semester < 1 || semester > 14) {
        throw ArgumentError('Semester harus antara 1-14, diberikan: $semester');
      }
      
      // Validasi nama tidak kosong
      if (nama.trim().isEmpty) {
        throw ArgumentError('Nama tidak boleh kosong');
      }
      
      // Cek duplikasi NIM
      if (_dataMahasiswa.containsKey(nim)) {
        throw StateError('Mahasiswa dengan NIM $nim sudah terdaftar');
      }
      
      // Tambah data jika semua validasi passed
      _dataMahasiswa[nim] = {
        'nama': nama.trim(),
        'jurusan': jurusan.trim(),
        'semester': semester,
        'ipk': ipk,
        'tanggalDaftar': DateTime.now(),
      };
      
      print('✅ Mahasiswa $nama berhasil ditambahkan dengan NIM $nim');
      
    } catch (e) {
      print('❌ Error menambah mahasiswa: $e');
      rethrow; // Re-throw untuk handling di level atas
    }
  }
  
  // Mencari mahasiswa dengan error handling
  Map cariMahasiswa(String nim) {
    try {
      if (!_dataMahasiswa.containsKey(nim)) {
        throw MahasiswaNotFoundException(nim);
      }
      
      return Map.from(_dataMahasiswa[nim]!);
      
    } on MahasiswaNotFoundException {
      print('❌ Mahasiswa dengan NIM $nim tidak ditemukan');
      rethrow;
    } catch (e) {
      print('❌ Error tidak terduga saat mencari mahasiswa: $e');
      rethrow;
    }
  }
  
  // Update IPK dengan validation
  void updateIPK(String nim, double ipkBaru) {
    try {
      // Cari mahasiswa dulu
      var mahasiswa = cariMahasiswa(nim);
      
      // Validasi IPK baru
      if (!_isValidIPK(ipkBaru)) {
        throw InvalidIPKException('IPK baru tidak valid: $ipkBaru');
      }
      
      double ipkLama = mahasiswa['ipk'];
      _dataMahasiswa[nim]!['ipk'] = ipkBaru;
      
      print('✅ IPK ${mahasiswa['nama']} berhasil diupdate dari $ipkLama ke $ipkBaru');
      
    } catch (e) {
      print('❌ Error update IPK: $e');
    }
  }
  
  // Hapus mahasiswa
  bool hapusMahasiswa(String nim) {
    try {
      var mahasiswa = cariMahasiswa(nim);
      _dataMahasiswa.remove(nim);
      print('✅ Mahasiswa ${mahasiswa['nama']} berhasil dihapus');
      return true;
      
    } catch (e) {
      print('❌ Error hapus mahasiswa: $e');
      return false;
    }
  }
  
  // Operasi bulk dengan error handling
  void tambahMahasiswaBulk(List> dataMahasiswa) {
    int berhasil = 0;
    int gagal = 0;
    
    for (var data in dataMahasiswa) {
      try {
        tambahMahasiswa(
          nim: data['nim'] ?? '',
          nama: data['nama'] ?? '',
          jurusan: data['jurusan'] ?? '',
          semester: data['semester'] ?? 0,
          ipk: data['ipk'] ?? 0.0,
        );
        berhasil++;
      } catch (e) {
        gagal++;
        print('Gagal menambah mahasiswa: ${data['nama']} - $e');
      }
    }
    
    print('\n📊 Hasil bulk insert: $berhasil berhasil, $gagal gagal');
  }
  
  // Display semua mahasiswa
  void tampilkanSemuaMahasiswa() {
    if (_dataMahasiswa.isEmpty) {
      print('Tidak ada data mahasiswa');
      return;
    }
    
    print('\n=== DAFTAR SEMUA MAHASISWA ===');
    _dataMahasiswa.forEach((nim, data) {
      print('NIM: $nim');
      print('Nama: ${data['nama']}');
      print('Jurusan: ${data['jurusan']}');
      print('Semester: ${data['semester']}');
      print('IPK: ${data['ipk']}');
      print('Tanggal Daftar: ${data['tanggalDaftar']}');
      print('---');
    });
  }
}

class MahasiswaAsyncManager {
  List> _cachedData = [];
  
  // Method dengan async/await
  Future loadMahasiswaData(String nim) async {
    try {
      print('\n=== LOADING MAHASISWA DATA ===');
      
      // Step 1: Validate NIM
      bool isValid = await ApiService.validateNIM(nim);
      if (!isValid) {
        throw Exception('NIM format tidak valid');
      }
      
      // Step 2: Fetch data dari database
      var mahasiswaData = await DatabaseService.fetchMahasiswa(nim);
      
      // Step 3: Cache data
      _cachedData.add(mahasiswaData);
      
      print('📦 Data mahasiswa berhasil dimuat dan di-cache');
      print('Data: ${mahasiswaData['nama']} - ${mahasiswaData['jurusan']}');
      
    } catch (e) {
      print('❌ Error loading mahasiswa data: $e');
    }
  }
  
  // Method dengan Future.wait untuk parallel processing
  Future loadMultipleMahasiswa(List nimList) async {
    try {
      print('\n=== PARALLEL LOADING MULTIPLE MAHASISWA ===');
      
      // Create list of futures
      List>> futures = nimList
          .map((nim) => DatabaseService.fetchMahasiswa(nim))
          .toList();
      
      // Wait for all futures to complete
      print('🔄 Memuat ${nimList.length} mahasiswa secara parallel...');
      
      List> results = await Future.wait(futures, eagerError: false);
      
      _cachedData.addAll(results);
      
      print('✅ Semua data berhasil dimuat dalam cache');
      print('Total data di cache: ${_cachedData.length}');
      
    } catch (e) {
      print('❌ Error dalam parallel loading: $e');
    }
  }
  
  // Method dengan timeout
  Future loadWithTimeout(String nim, {int timeoutSeconds = 3}) async {
    try {
      print('\n=== LOADING WITH TIMEOUT ===');
      
      var future = DatabaseService.fetchMahasiswa(nim);
      var result = await future.timeout(
        Duration(seconds: timeoutSeconds),
        onTimeout: () {
          throw TimeoutException('Request timeout setelah $timeoutSeconds detik');
        },
      );
      
      _cachedData.add(result);
      print('✅ Data loaded successfully within timeout');
      
    } on TimeoutException catch (e) {
      print('⏰ Timeout: $e');
    } catch (e) {
      print('❌ Error: $e');
    }
  }
  
  // Method dengan error handling dan retry logic
  Future?> loadWithRetry(String nim, {int maxRetries = 3}) async {
    int attempts = 0;
    
    while (attempts < maxRetries) {
      try {
        print('\n🔄 Attempt ${attempts + 1} untuk NIM $nim');
        
        var result = await DatabaseService.fetchMahasiswa(nim);
        print('✅ Berhasil pada attempt ${attempts + 1}');
        return result;
        
      } catch (e) {
        attempts++;
        print('❌ Attempt $attempts gagal: $e');
        
        if (attempts < maxRetries) {
          int delay = attempts * 1000; // Exponential backoff
          print('⏳ Retry dalam $delay ms...');
          await Future.delayed(Duration(milliseconds: delay));
        }
      }
    }
    
    print('💥 Semua attempts gagal untuk NIM $nim');
    return null;
  }
  
  // Method untuk async data processing
  Future> processAndSaveData(String nim) async {
    try {
      print('\n=== PROCESS AND SAVE DATA ===');
      
      // Step 1: Fetch data
      var data = await DatabaseService.fetchMahasiswa(nim);
      
      // Step 2: Process data (add computed fields)
      data['predikat'] = _hitungPredikat(double.parse(data['ipk']));
      data['statusKelulusan'] = _hitungStatusKelulusan(data['semester'], double.parse(data['ipk']));
      data['processedAt'] = DateTime.now().toIso8601String();
      
      // Step 3: Save processed data
      await DatabaseService.saveMahasiswa(data);
      
      // Step 4: Update cache
      _cachedData.add(data);
      
      return data;
      
    } catch (e) {
      print('❌ Error processing data: $e');
      rethrow;
    }
  }
  
  // Helper methods
  String _hitungPredikat(double ipk) {
    if (ipk >= 3.75) return 'Summa Cum Laude';
    if (ipk >= 3.50) return 'Magna Cum Laude';
    if (ipk >= 3.25) return 'Cum Laude';
    if (ipk >= 3.00) return 'Sangat Memuaskan';
    return 'Memuaskan';
  }
  
  String _hitungStatusKelulusan(int semester, double ipk) {
    if (semester >= 8 && ipk >= 2.0) return 'Lulus';
    if (semester >= 8 && ipk < 2.0) return 'Tidak Lulus';
    return 'Aktif';
  }
  
  // Display cached data
  void displayCachedData() {
    print('\n=== CACHED DATA ===');
    if (_cachedData.isEmpty) {
      print('Tidak ada data di cache');
      return;
    }
    
    for (var data in _cachedData) {
      print('${data['nama']} (${data['nim']}) - ${data['jurusan']} - IPK: ${data['ipk']}');
    }
  }
}

    if (weight < 10 || weight > 300) {
      throw InvalidWeightException('Berat badan harus antara 10-300 kg, diberikan: $weight kg');
    }
  }
  
  // Calculate BMI
  double calculateBMI() {
    double heightInMeter = _height / 100;
    return _weight / (heightInMeter * heightInMeter);
  }
  
  // Get BMI Category berdasarkan standar Indonesia
  BMICategory getBMICategory() {
    double bmi = calculateBMI();
    
    if (bmi < 17.0) return BMICategory.underweight;
    if (bmi < 23.0) return BMICategory.normal;      // Standar Asia
    if (bmi < 27.0) return BMICategory.overweight;
    if (bmi < 30.0) return BMICategory.obeseClass1;
    if (bmi < 35.0) return BMICategory.obeseClass2;
    return BMICategory.obeseClass3;
  }

    switch (getBMICategory()) {

        return 'Kelebihan Berat Badan (Overweight)';

      case BMICategory.obeseClass2:
        return 'Obesitas Kelas II';
      case BMICategory.obeseClass3:
        return 'Obesitas Kelas III (Morbid)';
    }

  @override
  double calculateBMR() {
    if (_gender == Gender.male) {
      return 10 * _weight + 6.25 * _height - 5 * _age + 5;
    } else {
      return 10 * _weight + 6.25 * _height - 5 * _age - 161;
    }
  }

  double calculateTDEE() {
    double bmr = calculateBMR();
    double activityMultiplier = _getActivityMultiplier();
    return bmr * activityMultiplier;
  }
  
  double _getActivityMultiplier() {
    switch (_activityLevel) {
      case ActivityLevel.sedentary:
        return 1.2;
      case ActivityLevel.lightlyActive:
        return 1.375;
      case ActivityLevel.moderatelyActive:
        return 1.55;

        return 1.725;

class BMICalculatorManager {
  Map _users = {};
  List> _populationData = [];
  
  // Add user
  void addUser(BMICalculator calculator) {
    _users[calculator.name] = calculator;
    _updatePopulationData();
    print('✅ User ${calculator.name} berhasil ditambahkan');
  }
  
  // Get user
  BMICalculator? getUser(String name) {
    return _users[name];
  }
  
  // Remove user
  bool removeUser(String name) {
    if (_users.containsKey(name)) {
      _users.remove(name);
      _updatePopulationData();
      print('✅ User $name berhasil dihapus');
      return true;
    }
    print('❌ User $name tidak ditemukan');
    return false;
  }
  
  // Update population statistics
  void _updatePopulationData() {
    _populationData.clear();
    
    _users.forEach((name, calculator) {
      _populationData.add({
        'name': name,
        'bmi': calculator.calculateBMI(),
        'category': calculator.getBMICategory(),
        'age': calculator.age,
        'gender': calculator.gender,
      });
    });
  }
  
  // Get population statistics
  Map getPopulationStatistics() {
    if (_populationData.isEmpty) {
      return {'message': 'Tidak ada data populasi'};
    }
    
    // BMI statistics
    List bmiValues = _populationData.map((data) => data['bmi'] as double).toList();
    double avgBMI = bmiValues.reduce((a, b) => a + b) / bmiValues.length;
    double maxBMI = bmiValues.reduce((a, b) => a > b ? a : b);
    double minBMI = bmiValues.reduce((a, b) => a < b ? a : b);
    
    // Category distribution
    Map categoryCount = {};
    for (BMICategory category in BMICategory.values) {
      categoryCount[category] = 0;
    }
    
    for (var data in _populationData) {
      BMICategory category = data['category'];
      categoryCount[category] = categoryCount[category]! + 1;
    }
    
    // Gender distribution
    int maleCount = _populationData.where((data) => data['gender'] == Gender.male).length;
    int femaleCount = _populationData.where((data) => data['gender'] == Gender.female).length;
    
    return {
      'total_users': _populationData.length,
      'bmi_statistics': {
        'average': double.parse(avgBMI.toStringAsFixed(2)),
        'maximum': double.parse(maxBMI.toStringAsFixed(2)),
        'minimum': double.parse(minBMI.toStringAsFixed(2)),
      },
      'category_distribution': categoryCount,
      'gender_distribution': {
        'male': maleCount,
        'female': femaleCount,
      },
    };
  }
  
  // Display population report
  void displayPopulationReport() {
    var stats = getPopulationStatistics();
    
    if (stats.containsKey('message')) {
      print(stats['message']);
      return;
    }
    
    print('\n' + '='*60);
    print('📊 LAPORAN STATISTIK POPULASI BMI');
    print('='*60);
    
    print('\n👥 RINGKASAN:');
    print('   Total Pengguna: ${stats['total_users']}');
    
    var bmiStats = stats['bmi_statistics'];
    print('\n📊 STATISTIK BMI:');
    print('   Rata-rata: ${bmiStats['average']}');
    print('   Tertinggi: ${bmiStats['maximum']}');
    print('   Terendah: ${bmiStats['minimum']}');
    
    var genderDist = stats['gender_distribution'];
    print('\n👫 DISTRIBUSI JENIS KELAMIN:');
    print('   Laki-laki: ${genderDist['male']} orang');
    print('   Perempuan: ${genderDist['female']} orang');
    
    var categoryDist = stats['category_distribution'] as Map;
    print('\n📈 DISTRIBUSI KATEGORI BMI:');
    categoryDist.forEach((category, count) {
      String categoryName = _getCategoryName(category);
      double percentage = (count / stats['total_users'] * 100);
      print('   $categoryName: $count orang (${percentage.toStringAsFixed(1)}%)');
    });
    
    print('\n' + '='*60);
  }
  
  String _getCategoryName(BMICategory category) {
    switch (category) {
      case BMICategory.underweight: return 'Kurus';
      case BMICategory.normal: return 'Normal';
      case BMICategory.overweight: return 'Kelebihan BB';
      case BMICategory.obeseClass1: return 'Obesitas I';
      case BMICategory.obeseClass2: return 'Obesitas II';
      case BMICategory.obeseClass3: return 'Obesitas III';
    }
  }
  
  // List all users
  void listAllUsers() {
    print('\n👤 DAFTAR SEMUA PENGGUNA:');
    if (_users.isEmpty) {
      print('Tidak ada pengguna terdaftar');

      String category = calculator.getBMIDescription();

Daftar Istilah dan Singkatan

Abstract Class
Class yang tidak dapat diinstansiasi secara langsung dan biasanya mengandung abstract methods yang harus diimplementasi oleh subclass.
Abstraction
Prinsip OOP yang menyembunyikan detail implementasi complex dan hanya menampilkan fitur essential kepada user.
Async/Await
Syntax untuk menangani operasi asynchronous dalam Dart, memungkinkan kode non-blocking tanpa callback hell.
BMR (Basal Metabolic Rate)
Jumlah kalori minimum yang dibutuhkan tubuh untuk menjalankan fungsi dasar saat istirahat.
Collections
Struktur data yang dapat menyimpan multiple elements, seperti List, Map, dan Set dalam Dart.
Constructor
Method khusus yang dipanggil saat object dibuat, digunakan untuk menginisialisasi properties object.
Encapsulation
Prinsip OOP yang menyembunyikan data internal class dan memberikan akses melalui public methods.
Enum
Tipe data khusus yang mendefinisikan sekumpulan konstanta yang berkaitan, seperti gender atau activity level.
Exception Handling
Mekanisme untuk menangani error atau kondisi exceptional yang terjadi saat runtime program.
Future
Object yang merepresentasikan hasil dari operasi asynchronous yang akan selesai di masa mendatang.
Getter/Setter
Methods khusus untuk mengakses (getter) dan memodifikasi (setter) private properties dengan kontrol tambahan.
Inheritance
Mekanisme OOP dimana class baru (child) dapat mewarisi properties dan methods dari class yang sudah ada (parent).
List
Collection ordered yang dapat menyimpan duplicate elements dan diakses menggunakan index.
Map
Collection yang menyimpan data dalam bentuk key-value pairs, dimana setiap key adalah unique.
Method Overriding
Praktik mendefinisikan ulang method dari parent class di child class dengan implementasi yang berbeda.
Named Constructor
Constructor tambahan dengan nama khusus yang memberikan cara alternatif untuk membuat object.
Null Safety
Fitur Dart yang mencegah null reference errors dengan explicit nullable types menggunakan operator ?.
Object
Instance dari class yang memiliki state (properties) dan behavior (methods) yang didefinisikan oleh class tersebut.
OOP (Object-Oriented Programming)
Paradigma pemrograman yang mengorganisir kode dalam bentuk objects yang memiliki properties dan methods.
Polymorphism
Kemampuan objects dari different classes untuk merespon method call yang sama dengan cara yang berbeda.
Set
Collection yang hanya menyimpan unique elements tanpa duplicate dan tidak memiliki order yang tetap.
Static Method
Method yang belongs to class rather than instance, dapat dipanggil tanpa membuat object dari class.
TDEE (Total Daily Energy Expenditure)
Total kalori yang dibakar tubuh dalam sehari termasuk BMR dan aktivitas fisik.
Try-Catch Block
Konstruksi untuk menangani exceptions, dimana kode yang mungkin error diletakkan di try, dan handling di catch.
